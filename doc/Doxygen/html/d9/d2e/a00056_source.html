132 template <class fieldType, class nodeType>
133 void Foam::moment<fieldType, nodeType>::update()
134 {
135  // Resetting the moment to zero
136  *this == dimensionedScalar("moment", (*this).dimensions(), 0);
137 
138  const PtrList<nodeType>& nodes = nodes_();
139 
140  bool extendedNode = nodes[0].extended();
141 
142  // If nodes do not have extended status, only use primary quadrature.
143  if (!extendedNode)
144  {
145  forAll(nodes, pNodeI)
146  {
147  const nodeType& node = nodes[pNodeI];
148 
149  if (!node.extended())
150  {
151  fieldType m = node.primaryWeight();
152 
153  for (label cmpt = 0; cmpt < nDimensions_; cmpt++)
154  {
155  const label cmptMomentOrder = cmptOrders()[cmpt];
156 
157  tmp<fieldType> abscissaCmpt
158  = node.primaryAbscissa().component(cmpt);
159 
160  tmp<fieldType> mPow = m*pow(abscissaCmpt, cmptMomentOrder);
161  m.dimensions().reset(mPow().dimensions());
162 
163  m == mPow;
164  }
165 
166  *this == *this + m;
167  }
168  }
169 
170  return;
171  }
172 
173  // Extended quadrature case
174  forAll(nodes, pNodeI)
175  {
176  const nodeType& node = nodes[pNodeI];
177 
178  const fieldType& pW = node.primaryWeight();
179 
180  for (label sNodeI = 0; sNodeI < node.nSecondaryNodes(); sNodeI++)
181  {
182  fieldType m(pW*node.secondaryWeights()[sNodeI]);
183 
184  for (label cmpt = 0; cmpt < nDimensions_; cmpt++)
185  {
186  const label cmptMomentOrder = cmptOrders()[cmpt];
187 
188  tmp<fieldType> abscissaCmpt
189  = node.secondaryAbscissae()[sNodeI].component(cmpt);
190 
191  tmp<fieldType> mPow = m*pow(abscissaCmpt, cmptMomentOrder);
192 
193  m.dimensions().reset(mPow().dimensions());
194 
195  m == mPow;
196  }
197 
198  *this == *this + m;
199  }
200  }
201 }
